Future<void> openVideoPlayerWithSource({
  required OpenVideoPlayerSource source,
  OpenVideoPlayerConfiguration? config,
}) async {
  if (!FWGlobalState.getInstance().sdkInitSucceeded) {
    await FWGlobalState.getInstance().sdkInitSucceededFuture;
  }
  final nativeMethodName =
      FireworkSDKNativeMethodName.openVideoPlayerWithSource.name;
  final arg = source.toMap();
  arg["config"] = VideoPlayerConfigurationUtil.toJson(config);

  FWMethodChannelUtil.getFirewokSDKChannel()
      .invokeMethod(nativeMethodName, arg);
}
