set cartIconVisible(bool cartIconVisible) {
  _cartIconVisible = cartIconVisible;
  final nativeMethodName = ShoppingNativeMethodName.setCartIconVisible.name;
  if (FWGlobalState.getInstance().sdkInitSucceeded) {
    FWMethodChannelUtil.getShoppingChannel().invokeMethod(
      nativeMethodName,
      cartIconVisible,
    );
  } else {
    FWGlobalState.getInstance().sdkInitSucceededFuture.then((_) {
      FWMethodChannelUtil.getShoppingChannel().invokeMethod(
        nativeMethodName,
        cartIconVisible,
      );
    });
  }
}
